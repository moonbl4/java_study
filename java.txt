        int minute = now.get(ChronoField.MINUTE_OF_HOUR);
        int month = now.get(ChronoField.MONTH_OF_YEAR);
        int day = now.get(ChronoField.DAY_OF_MONTH);
        int year = now.get(ChronoField.YEAR);

        if (hour < 12) {
            System.out.println("Доброе утро!\n");

        } else {
            System.out.println("Добрый вечер!\n");

        System.out.print("Сейчас ");

        //Сколько часовъ
        System.out.print(hour);
        System.out.print((hour != 1 & hour != 21)
                ? " часа (часов) " : " час ");
        //Сколько минут
        if (minute != 0) {
            System.out.print(minute);
            System.out.print((minute != 1 & minute != 21 & minute != 31 & minute != 41 & minute != 51)
                    ? " минуты (минут)" : " минута");
        }
        //Отображения числа месяца
        System.out.print("\n" + day + " ");

        //Название месяца

Для массивов существует несколько дополнительных методов, что позволяют проводить действия над массивом. Рассмотрим несколько из них: 

length - возвращает количество элементов в массиве. К примеру, создадим массив arr и укажем для него 3 элемента. При вызове функции будет выдано число 3: arr.length;

Arrays.copyOf - копирует один массив в другой. 
Пример:
int[] arr = new int[] { 1, 2, 3, 5 };
int[] some_new = Arrays.copyOf(arr, arr.length);

Arrays.sort - сортировка массива.
Пример:
int[] numbers = new int[] { 1, 7, 3, 5, 2, 6, 4 };
 
Arrays.sort(numbers);//
